API Requests

The following examples use the JSON API Specification. YMMV.

Basic Search

# PUT /people_search/:id

{
  data: {
    attributes: {
      conditions: {
        { name: "Bart" }
      }
    }
  }
}

Response

{
  data: {
    id: '1dc9c145-80cb-4d39-801d-1189afe1ec8c',
    type: 'people-searches',
    attributes: {
      conditions: {
        { name: "Bart" }
      }
    },
    relationships: {
      results: {
        data: [
          { id: 123, type: 'people' }
        ]
      }
    },
    included: [
      { id: 123, type: 'people', attributes: { name: 'Bart' } }
    ]
  }
}
